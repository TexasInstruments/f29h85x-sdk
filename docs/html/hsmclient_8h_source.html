319typedef struct OTFA_Region_t
320{
321 uint8_t authMode ; /* mode of authentication - disable-0/GMAC-1/CMAC-2 ; */
322 uint8_t encMode ; /* mode of decryption - disable-0 or AES_CTR-1 */
323 uint16_t reservedArea ; /* reserved to align with 4kB structure */
324 uint32_t regionStAddr ; /* start address of the flash region for which the configuration should apply */
325 uint32_t regionSize ; /* size of the flash region in kB for which the configuration should apply */
326 uint8_t authKeyID ; /* Keyring ID of key to be used for authentication */
327 uint8_t encrKeyID ; /* Keyring ID of key to be used for encryption */
328 uint8_t encrKeyFetchMode ; /* specify which 16 bytes of DSMEK are to be used - 1 for fist 16/2 for last 16/ 3 for XOR of both */
329 uint8_t authAesKey [16] ; /* actual key value to be written to the register for authentication ; fetched from keyring */
330 uint8_t encrAesKey [16] ; /* actual key value to be written to the register for decryption ; fetched from keyring */
331 uint8_t regionIV[16] ; /* IV to be used for encryption */
332}OTFA_Region_t ;
