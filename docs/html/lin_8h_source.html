588 //*****************************************************************************
589 //
602 //
603 //*****************************************************************************
604 static inline uint16_t
605 LIN_generateParityID(uint16_t identifier)
606 {
607 
608  uint16_t p0, p1, parityIdentifier;
609 
610  //
611  // Calculate parity bits and generate updated identifier
612  //
613  p0 = ((identifier & LIN_ID0) ^ ((identifier & LIN_ID1) >> 1U) ^
614  ((identifier & LIN_ID2) >> 2U) ^ ((identifier & LIN_ID4) >> 4U));
615  p1 = (uint16_t)(!(((identifier & LIN_ID1) >> 1U)
616  ^ ((identifier & LIN_ID3) >> 3U)
617  ^ ((identifier & LIN_ID4) >> 4U)
618  ^ ((identifier & LIN_ID5) >> 5U)));
619  parityIdentifier = identifier | ((p0 << 6U) | (p1 << 7U));
620 
621  return(parityIdentifier);
622 }
