578 __attribute__((always_inline))
579 static inline uint16_t
580 LIN_generateParityID(uint16_t identifier)
581 {
582 
583  uint16_t p0, p1, parityIdentifier;
584 
585  //
586  // Calculate parity bits and generate updated identifier
587  //
588  p0 = ((identifier & LIN_ID0) ^ ((identifier & LIN_ID1) >> 1U) ^
589  ((identifier & LIN_ID2) >> 2U) ^ ((identifier & LIN_ID4) >> 4U));
590  p1 = (uint16_t)(!(bool)(((identifier & LIN_ID1) >> 1U)
591  ^ ((identifier & LIN_ID3) >> 3U)
592  ^ ((identifier & LIN_ID4) >> 4U)
593  ^ ((identifier & LIN_ID5) >> 5U)));
594  parityIdentifier = identifier | ((p0 << 6U) | (p1 << 7U));
595 
596  return(parityIdentifier);
597 }
