212static inline void
213DCC_enableModule(uint32_t base)
214{
215 //
216 // Check the arguments.
217 //
218 ASSERT(DCC_isBaseValid(base));
219
220 //
221 // Set DCC enable bit field.
222 //
223 HWREGH(base + DCC_O_GCTRL) =
224 (HWREGH(base + DCC_O_GCTRL) & ~(DCC_GCTRL_DCCENA_M)) |
225 ((uint16_t)DCC_ENABLE_VALUE << DCC_GCTRL_DCCENA_S);
226}

616static inline void
617DCC_setCounter1ClkSource(uint32_t base, DCC_Count1ClockSource source)
618{
619 //
620 // Check the arguments.
621 //
622 ASSERT(DCC_isBaseValid(base));
623
624 //
625 // Set the specified clock source
626 // DCC Clk source is of 5bits DCCCLKSRC1[4:0]
627 //
628 HWREGH(base + DCC_O_CLKSRC1) = (HWREGH(base + DCC_O_CLKSRC1) &
629 (DCC_CLKSRC1_CLKSRC1_M << 6U)) |
630 (((uint16_t)DCC_ENABLE_VALUE <<
631 DCC_CLKSRC1_KEY_S) | (uint16_t)source);
632}

736static inline void
737DCC_setCounterSeeds(uint32_t base, uint32_t counter0, uint32_t validCounter0,
738 uint32_t counter1)
739{
740 //
741 // Check the arguments.
742 //
743 ASSERT(DCC_isBaseValid(base));
744 ASSERT(validCounter0 >= DCC_VALIDSEED_MIN);
745 ASSERT(counter0 >= DCC_COUNTSEED_MIN);
746 ASSERT(counter1 >= DCC_COUNTSEED_MIN);
747 ASSERT(validCounter0 <= DCC_VALIDSEED_MAX);
748 ASSERT(counter0 <= DCC_COUNTSEED_MAX);
749 ASSERT(counter1 <= DCC_COUNTSEED_MAX);
750
751 //
752 // Set Counter 0 Seed
753 //
754 HWREG(base + DCC_O_CNTSEED0) = counter0;
755
756 //
757 // Set Valid Duration Counter 0 Seed
758 //
759 HWREG(base + DCC_O_VALIDSEED0) = validCounter0;
760
761 //
762 // Set Counter 1 Seed
763 //
764 HWREG(base + DCC_O_CNTSEED1) = counter1;
765
766}
