279 {

426 static inline uint32_t
427 GPIO_readPin(uint32_t pin)
428 {
429  volatile uint32_t *gpioDataReg;
430 
431  //
432  // Check the arguments.
433  //
434  ASSERT(GPIO_isPinValid(pin));
435 
436  gpioDataReg = (uint32_t *)((uintptr_t)GPIODATA_BASE) +
437  ((pin / 32U) * GPIO_DATA_REGS_STEP);
438 
439  return((gpioDataReg[GPIO_GPxDAT_INDEX] >> (pin % 32U)) & (uint32_t)0x1U);
440 }

491 static inline void
492 GPIO_writePin(uint32_t pin, uint32_t outVal)
493 {
494  volatile uint32_t *gpioDataReg;
495  uint32_t pinMask;
496 
497  //
498  // Check the arguments.
499  //
500  ASSERT(GPIO_isPinValid(pin));
501 
502  gpioDataReg = (uint32_t *)((uintptr_t)GPIODATA_BASE) +
503  ((pin / 32U) * GPIO_DATA_REGS_STEP);
504 
505  pinMask = (uint32_t)1U << (pin % 32U);
506 
507  if(outVal == 0U)
508  {
509  gpioDataReg[GPIO_GPxCLEAR_INDEX] = pinMask;
510  }
511  else
512  {
513  gpioDataReg[GPIO_GPxSET_INDEX] = pinMask;
514  }
515 }
