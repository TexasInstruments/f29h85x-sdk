554//*****************************************************************************
555//
563//
564//*****************************************************************************
565static inline uint64_t
566IPC_getCounter(void)
567{
568 //
569 // Get the Counter High and Low values. Read to the Counter low register
570 // saves the value of Counter High register.
571 //
572 uint32_t ctrL = HWREG(IPCCOUNTER_BASE + IPC_O_COUNTERL);
573 uint32_t ctrH = HWREG(IPCCOUNTER_BASE + IPC_O_COUNTERH);
574
575 //
576 // Return the 64-bit value of the counter
577 //
578 return(((uint64_t)ctrH << 32U) | ((uint64_t)ctrL));
579}
