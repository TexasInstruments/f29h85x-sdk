529 static inline uint64_t
530 IPC_getCounter(void)
531 {
532  //
533  // Get the Counter High and Low values. Read to the Counter low register
534  // saves the value of Counter High register.
535  //
536  uint32_t ctrL = HWREG(IPCCOUNTERREGS_BASE + IPC_O_COUNTERL);
537  uint32_t ctrH = HWREG(IPCCOUNTERREGS_BASE + IPC_O_COUNTERH);
538 
539  //
540  // Return the 64-bit value of the counter
541  //
542  return(((uint64_t)ctrH << 32U) | ((uint64_t)ctrL));
543 }
