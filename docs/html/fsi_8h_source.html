533FSI_sendTxFlush(uint32_t base)
534{
535 //
536 // Check the arguments.
537 //
538 ASSERT(FSI_isTxBaseValid(base));
539
540 HWREGH(base + FSI_O_TX_MAIN_CTRL) |= ((uint16_t)FSI_TX_MAIN_CTRL_FLUSH |
541 (FSI_CTRL_REG_KEY <<
542 FSI_TX_MAIN_CTRL_KEY_S));
543}

569 FSI_TX_MAIN_CTRL_KEY_S));
