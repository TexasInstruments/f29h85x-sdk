452//*****************************************************************************

477//
478//*****************************************************************************
479static inline void
480EQEP_setDecoderConfig(uint32_t base, uint16_t config)
481{
482 //
483 // Check the arguments.
484 //
485 ASSERT(EQEP_isBaseValid(base));
486
487 //
488 // Write the new decoder configuration to the hardware.
489 //
490 HWREGH(base + EQEP_O_QDECCTL) = (HWREGH(base + EQEP_O_QDECCTL) &
491 ~(EQEP_QDECCTL_SWAP |
492 EQEP_QDECCTL_XCR |
493 EQEP_QDECCTL_QSRC_M |
494 EQEP_QDECCTL_IGATE)) | config;
495}

611EQEP_getDirection(uint32_t base)
612{
613 int16_t direction;
614
615 //
616 // Check the arguments.
617 //
618 ASSERT(EQEP_isBaseValid(base));
619
620 //
621 // Return the direction of rotation.
622 //
623 if((HWREGH(base + EQEP_O_QEPSTS) & EQEP_QEPSTS_QDF) != 0U)

627 else
628 {
