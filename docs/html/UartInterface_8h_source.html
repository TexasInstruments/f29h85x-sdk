54 void configComPort(const std::string& portName, const size_t baudRate = 115200)
55 {
56 try
57 {

60 }
61 catch (const std::exception& e)
62 {
63 std::string updatedMessage = "ERROR: please double check COM/tty port, does it exist? \nSystem message: " + std::string(e.what());
64 throw std::runtime_error(updatedMessage);
65 }
66
67 try
68 {
69 _uartPort.set_option(asio::serial_port::baud_rate(baudRate));
70 _uartPort.set_option(asio::serial_port::character_size(8));
71 _uartPort.set_option(asio::serial_port::stop_bits(asio::serial_port::stop_bits::one));
72 _uartPort.set_option(asio::serial_port::flow_control(asio::serial_port::flow_control::none));
73 _uartPort.set_option(asio::serial_port::parity(asio::serial_port::parity::none));
