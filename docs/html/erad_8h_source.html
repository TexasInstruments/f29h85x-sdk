584 __attribute__((always_inline))
585 static inline void
586 ERAD_setOwnership(ERAD_Owner owner)
587 {
588  uint8_t instance = 0;
589  //
590  // Set the Owner of all EBC instances as owner
591  //
592  for (instance = 0; instance < ERAD_MAX_EBC_INSTANCES; instance++)
593  {
594  HWREGB(ERAD_BASE + ERAD_O_EBC_OWNER((uint32_t)instance)) =
595  ((uint8_t)owner & ERAD_EBC_OWNER_OWNER_M);
596  }
597 
598  //
599  // Set the Owner of all SEC instances as owner
600  //
601  for (instance = 0; instance < ERAD_MAX_SEC_INSTANCES; instance++)
602  {
603  HWREGB(ERAD_BASE + ERAD_O_SEC_OWNER((uint32_t)instance)) =
604  ((uint8_t)owner & ERAD_SEC_OWNER_OWNER_M);
605  }
606 
607  //
608  // Set the Owner of all AND Mask instances as owner
609  //
610  for (instance = 0; instance < ERAD_MAX_AND_MASK_INSTANCES; instance++)
611  {
612  HWREGB(ERAD_BASE + ERAD_O_AND_MASK_OWNER((uint32_t)instance)) =
613  ((uint8_t)owner & ERAD_AND_MASK_OWNER_OWNER_M);
614  }
615 
616  //
617  // Set the Owner of all OR Mask instances as owner
618  //
619  for (instance = 0; instance < ERAD_MAX_OR_MASK_INSTANCES; instance++)
620  {
621  HWREGB(ERAD_BASE + ERAD_O_OR_MASK_OWNER((uint32_t)instance)) =
622  ((uint8_t)owner & ERAD_OR_MASK_OWNER_OWNER_M);
623  }
624 
625  //
626  // Set the Owner of the PC trace module as owner
627  //
628  HWREGB(ERAD_BASE + ERAD_O_PCTRACE_OWNER) =
629  ((uint8_t)owner & ERAD_PCTRACE_OWNER_OWNER_M);
630 }
