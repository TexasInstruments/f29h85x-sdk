616static inline void
617ERAD_setOwnership(ERAD_Owner owner)
618{
619 uint8_t instance = 0;
620 //
621 // Set the Owner of all EBC instances as owner
622 //
623 for (instance = 0; instance < ERAD_MAX_EBC_INSTANCES; instance++)
624 {
625 HWREGB(ERAD_BASE + ERAD_O_EBC_OWNER((uint32_t)instance)) =
626 ((uint8_t)owner & ERAD_EBC_OWNER_OWNER_M);
627 }
628
629 //
630 // Set the Owner of all SEC instances as owner
631 //
632 for (instance = 0; instance < ERAD_MAX_SEC_INSTANCES; instance++)
633 {
634 HWREGB(ERAD_BASE + ERAD_O_SEC_OWNER((uint32_t)instance)) =
635 ((uint8_t)owner & ERAD_SEC_OWNER_OWNER_M);
636 }
637
638 //
639 // Set the Owner of all AND Mask instances as owner
640 //
641 for (instance = 0; instance < ERAD_MAX_AND_MASK_INSTANCES; instance++)
642 {
643 HWREGB(ERAD_BASE + ERAD_O_AND_MASK_OWNER((uint32_t)instance)) =
644 ((uint8_t)owner & ERAD_AND_MASK_OWNER_OWNER_M);
645 }
646
647 //
648 // Set the Owner of all OR Mask instances as owner
649 //
650 for (instance = 0; instance < ERAD_MAX_OR_MASK_INSTANCES; instance++)
651 {
652 HWREGB(ERAD_BASE + ERAD_O_OR_MASK_OWNER((uint32_t)instance)) =
653 ((uint8_t)owner & ERAD_OR_MASK_OWNER_OWNER_M);
654 }
655
656 //
657 // Set the Owner of the PC trace module as owner
658 //
659 HWREGB(ERAD_BASE + ERAD_O_PCTRACE_OWNER) =
660 ((uint8_t)owner & ERAD_PCTRACE_OWNER_OWNER_M);
661}
