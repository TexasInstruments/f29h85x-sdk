616 static inline void
617 ERAD_setOwnership(ERAD_Owner owner)
618 {
619  uint8_t instance = 0;
620  //
621  // Set the Owner of all EBC instances as owner
622  //
623  for (instance = 0; instance < ERAD_MAX_EBC_INSTANCES; instance++)
624  {
625  HWREGB(ERAD_BASE + ERAD_O_EBC_OWNER((uint32_t)instance)) =
626  ((uint8_t)owner & ERAD_EBC_OWNER_OWNER_M);
627  }
628 
629  //
630  // Set the Owner of all SEC instances as owner
631  //
632  for (instance = 0; instance < ERAD_MAX_SEC_INSTANCES; instance++)
633  {
634  HWREGB(ERAD_BASE + ERAD_O_SEC_OWNER((uint32_t)instance)) =
635  ((uint8_t)owner & ERAD_SEC_OWNER_OWNER_M);
636  }
637 
638  //
