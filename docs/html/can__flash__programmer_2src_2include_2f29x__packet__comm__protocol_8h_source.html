54 /* SBL response content to the host command */
55 typedef struct CommandResponsePkt_t_
56 {
57 uint8_t payloadType; // Must be enum PacketType_ CommandResponse
58 uint16_t echoCommand; // Reply with the same command if recognized, if unrecognized, use COMMAND_NOT_RECOGNIZED
59 uint8_t expectsData; // Does SBL expects further data for this command, 0 for false or otherwise true (boolean value)
60 uint16_t timeDelay_ms; // Host time delay before subsequent data transmition, in ms, irrelevant if "expectsData" is false.
61 uint16_t maxPayloadSize; // Maximum size of data SBL can parse at once, irrelvant if "expectsData" is false, 0 will use host default setting
62 uint8_t expectsEOTpacket; // Would SBL sends out an addintional packet at end-of-transmission to the host, 0 for false or otherwise true (boolean value)

64 } CommandResponsePkt_t;
