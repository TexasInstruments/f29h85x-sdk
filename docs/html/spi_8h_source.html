505SPI_setcharLength(uint32_t base, uint16_t charLength)
506{

508 SPI_disableModule(base);
509 HWREGH(base + SPI_O_CCR) = (HWREGH(base + SPI_O_CCR) &

511 (charLength - 1U);
512 SPI_enableModule(base);

642 SPI_RxFIFOLevel rxLevel)

653 (uint16_t)(~(SPI_FFTX_TXFFIL_M))) | (uint16_t)txLevel;
654 HWREGH(base + SPI_O_FFRX) = (HWREGH(base + SPI_O_FFRX) &
655 (uint16_t)(~(SPI_FFRX_RXFFIL_M))) | (uint16_t)rxLevel;
656}

658//*****************************************************************************
659//
674//
675//*****************************************************************************
676static inline void
677SPI_getFIFOInterruptLevel(uint32_t base, SPI_TxFIFOLevel *txLevel,
678 SPI_RxFIFOLevel *rxLevel)
679{
680 //
681 // Check the arguments.
682 //
683 ASSERT(SPI_isBaseValid(base));
684
685 //
686 // Extract the transmit and receive FIFO levels.
687 //
688 *txLevel = (SPI_TxFIFOLevel)((uint16_t)(HWREGH(base + SPI_O_FFTX) &
689 SPI_FFTX_TXFFIL_M));
690 *rxLevel = (SPI_RxFIFOLevel)((uint16_t)(HWREGH(base + SPI_O_FFRX) &
691 SPI_FFRX_RXFFIL_M));
692}
