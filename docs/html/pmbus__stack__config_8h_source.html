1184static bool
1185PMBusStackObject_isCommandAndTransactionValid(const uint16_t command,
1186 const PMBus_Transaction transaction)
1187{
1188 uint16_t base, downshift;
1189 PMBus_Transaction transactionNibble;
1190 bool status = false;
1191
1192 //
1193 // The base address in the commandMap is command/4
1194 //
1195 base = (command >> 2U) & BASE_ADDRESS_MASK;
1196
1197 //
1198 // The transaction is either field0, 1, 2, or 3, we are
1199 // going to determine how much right shift we need to have
1200 // our reference transaction in the lowest nibble. Since
1201 // each field is 4 bits, we multiply by 4
1202 //
1203 downshift = (command & COMMAND_MASK) << 2U;
1204
1205 //
1206 // Find the transaction nibble in the commandMap
1207 //
1208 transactionNibble = (PMBus_Transaction)
1209 ((PMBusStack_commandTransactionMap[base].transactionField >>
1210 downshift) & TRANSACTION_NIBBLE_MASK);
1211
1212 if(transactionNibble == transaction)
1213 {
1214 status = true;
1215 }
1216
1217 return(status);
1218}
