1055 ADC_Channel channel, uint32_t sampleWindow)
1056{

1058
1059 //
1060 // Check the arguments.
1061 //

1063 ASSERT((sampleWindow >= 1U) && (sampleWindow <= 512U));
1064
1065 mask = (ADC_SOC0CTL_CHSEL_M | ADC_SOC0CTL_TRIGSEL_M | ADC_SOC0CTL_ACQPS_M);

1067 //
1068 // Calculate address for the SOC control register.
1069 //
1070 ctlRegAddr = base + ADC_SOCxCTL_OFFSET_BASE + ((uint32_t)socNumber * 4U);
1071
1072 //
1073 // Set the configuration of the specified SOC.
1074 //
1075
1076 HWREG(ctlRegAddr) = (HWREG(ctlRegAddr) & ~(mask)) |
1077 ((uint32_t)channel << ADC_SOC0CTL_CHSEL_S) |
1078 ((uint32_t)trigger << ADC_SOC0CTL_TRIGSEL_S) |
1079 (sampleWindow - 1U);
1080}

1245 ADC_IntSOCTrigger trigger)
1246{
1247 uint16_t shiftVal;
1248
1249 //
1250 // Check the arguments.
1251 //
1252 ASSERT(ADC_isBaseValid(base));
1253
1254 uint32_t regOffset = ADC_O_INTSOCSEL1;
1255
1256 if(socNumber >= ADC_SOC_NUMBER16)

1258 socNumber = (ADC_SOCNumber)((uint32_t)socNumber - (uint32_t)ADC_SOC_NUMBER16);

1262 // Each SOC has a 2-bit field in this register.
1263 //
1264 shiftVal = (uint16_t)socNumber << 1U;

1267 // Set the configuration of the specified SOC. Not that we're treating
1268 // ADCINTSOCSEL1 and ADCINTSOCSEL2 as one 32-bit register here.
1269 //
1270
1271 HWREG(base + regOffset) = (HWREG(base + regOffset) &
1272 ~((uint32_t)ADC_INTSOCSEL1_SOC0_M <<
1273 shiftVal)) |
1274 ((uint32_t)trigger << shiftVal);
1275
1276}
