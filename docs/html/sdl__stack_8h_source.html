212static inline SDL_Stack_Status SDL_Stack_getBusCompStatus(SDL_Stack_BusCompInstance instance)
213{
214 /* Read and return status of Enhanced Bus comparator (EBC) */
215 uint32_t regVal = ((HW_RD_REG32(SDL_ERAD_BASE +

218 return (SDL_Stack_Status) regVal;
