1242//*****************************************************************************
1243//
1254//
1255//*****************************************************************************
1256static inline void
1257WADI_configSSSTriggerSequence(uint32_t base,
1258 WADI_Event startEvent,
1259 WADI_Event endEvent)
1260{
1261 uint32_t i = 0U;
1262
1263 //
1264 // Check the arguments
1265 //
1266 ASSERT(WADI_isSSSBaseValid(base));
1267
1268 //
1269 // Converting the bitmask to event index
1270 //
1271 uint32_t temp;
1272 uint32_t indexStartEvent = 0;
1273 temp = (uint32_t)startEvent >> 1;
1274 while (temp != 0U)
1275 {
1276 indexStartEvent++;
1277 temp >>= 1;
1278 }
1279
1280 uint32_t indexEndEvent = 0;
1281 temp = (uint32_t)endEvent >> 1;
1282 while (temp != 0U)
1283 {
1284 indexEndEvent++;
1285 temp >>= 1;
1286 }
1287
1288 ASSERT(indexStartEvent <= indexEndEvent);
1289
1290 //
1291 // Set trigger types for the events
1292 //
1293 uint8_t trigType[8] = {0};
1294 for (i = 0; i <= 7U; i++)
1295 {
1296 if((i < indexStartEvent) || (i > indexEndEvent))
1297 {
1298 trigType[i] = 0x0U;
1299 }
1300 else if(i == indexEndEvent)
1301 {
1302 trigType[i] = 0x1U;
1303 }
1304 else
1305 {
1306 trigType[i] = 0x9U;
1307 }
1308 }
1309
1310 uint32_t reg1 = 0;
1311 uint32_t reg2 = 0;
1312
1313 //
1314 // The complement must be written before the actual value
1315 //
1316 for (i = 0; i < 4U; i++)
1317 {
1318 uint8_t val = trigType[i] & 0xFU;
1319 uint8_t byte = ((~val & 0xFU) << 4U) | val;
1320 reg1 |= ((uint32_t)byte << ((i % 4U) * 8U));
1321 }
1322
1323 if(indexEndEvent >= 4U)
1324 {
1325 for (i = 4U; i < 8U; i++)
1326 {
1327 uint8_t val = trigType[i] & 0xFU;
1328 uint8_t byte = ((~val & 0xFU) << 4U) | val;
1329 reg2 |= ((uint32_t)byte << ((i % 4U) * 8U));
1330 }
1331 }
1332 HWREG(base + WADI_O_SSS_TRIGEVT1_4CFG) |= reg1;
1333 HWREG(base + WADI_O_SSS_TRIGEVT5_8CFG) |= reg2;
1334}
