372//*****************************************************************************

382//*****************************************************************************
383static inline void
384I2C_enableFIFO(uint32_t base)
385{
386 //
387 // Check the arguments.
388 //
389 ASSERT(I2C_isBaseValid(base));
390
391 //
392 // Enable the FIFO.
393 //
394 HWREGH(base + I2C_O_FFTX) |= I2C_FFTX_I2CFFEN | I2C_FFTX_TXFFRST;
395 HWREGH(base + I2C_O_FFRX) |= I2C_FFRX_RXFFRST;
396}

409static inline void
410I2C_disableFIFO(uint32_t base)
411{
412 //
413 // Check the arguments.
414 //
415 ASSERT(I2C_isBaseValid(base));
416
417 //
418 // Disable the FIFO.
419 //
420 HWREGH(base + I2C_O_FFTX) &= ~(I2C_FFTX_I2CFFEN | I2C_FFTX_TXFFRST);
421 HWREGH(base + I2C_O_FFRX) &= ~I2C_FFRX_RXFFRST;
422}

424//*****************************************************************************
425//
443//
444//*****************************************************************************
445static inline void
446I2C_setFIFOInterruptLevel(uint32_t base, I2C_TxFIFOLevel txLevel,
447 I2C_RxFIFOLevel rxLevel)
448{
449 //
450 // Check the arguments.
451 //
452 ASSERT(I2C_isBaseValid(base));
453
454 //
455 // Set the FIFO interrupt levels.
456 //
457 HWREGH(base + I2C_O_FFTX) = (HWREGH(base + I2C_O_FFTX) &
458 (~I2C_FFTX_TXFFIL_M)) | (uint16_t)txLevel;
459 HWREGH(base + I2C_O_FFRX) = (HWREGH(base + I2C_O_FFRX) &
460 (~I2C_FFRX_RXFFIL_M)) | (uint16_t)rxLevel;
461}

463//*****************************************************************************
464//
482//
483//*****************************************************************************
484static inline void
485I2C_getFIFOInterruptLevel(uint32_t base, I2C_TxFIFOLevel *txLevel,
486 I2C_RxFIFOLevel *rxLevel)
487{
488 //
489 // Check the arguments.
490 //
491 ASSERT(I2C_isBaseValid(base));
492
493 //
494 // Extract the transmit and receive FIFO levels.
495 //
496 *txLevel = (I2C_TxFIFOLevel)(HWREGH(base + I2C_O_FFTX) &
497 I2C_FFTX_TXFFIL_M);
498 *rxLevel = (I2C_RxFIFOLevel)(HWREGH(base + I2C_O_FFRX) &
499 I2C_FFRX_RXFFIL_M);
500}
