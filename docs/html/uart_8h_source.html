118// These are 8 bit (1 byte) wide and 16 locations deep FIFOs.

274static inline void
275UART_setParityMode(uint32_t base, uint32_t parity)
276{
277 //
278 // Check the arguments.
279 //
280 ASSERT(UART_isBaseValid(base));
281 ASSERT((parity == UART_CONFIG_PAR_NONE) ||
282 (parity == UART_CONFIG_PAR_EVEN) ||
283 (parity == UART_CONFIG_PAR_ODD) ||
284 (parity == UART_CONFIG_PAR_ONE) ||
285 (parity == UART_CONFIG_PAR_ZERO));
286
287 //
288 // Set the parity mode.
289 //
290 HWREG(base + UART_O_LCRH) = ((HWREG(base + UART_O_LCRH) &
291 ~(UART_LCRH_SPS | UART_LCRH_EPS |
292 UART_LCRH_PEN)) | parity);
293}

326//*****************************************************************************
327//
342//
343//*****************************************************************************
344static inline void
345UART_setFIFOLevel(uint32_t base, uint32_t txLevel,
346 uint32_t rxLevel)
347{
348 //
349 // Check the arguments.
350 //
351 ASSERT(UART_isBaseValid(base));
352 ASSERT((txLevel == UART_FIFO_TX1_8) ||
353 (txLevel == UART_FIFO_TX2_8) ||
354 (txLevel == UART_FIFO_TX4_8) ||
355 (txLevel == UART_FIFO_TX6_8) ||
356 (txLevel == UART_FIFO_TX7_8));
357 ASSERT((rxLevel == UART_FIFO_RX1_8) ||
358 (rxLevel == UART_FIFO_RX2_8) ||
359 (rxLevel == UART_FIFO_RX4_8) ||
360 (rxLevel == UART_FIFO_RX6_8) ||
361 (rxLevel == UART_FIFO_RX7_8));
362
363 //
364 // Set the FIFO interrupt levels.
365 //
366 HWREGB(base + UART_O_IFLS) = (uint8_t)(txLevel | rxLevel);
367}

369//*****************************************************************************
370//
385//
386//*****************************************************************************
387static inline void
388UART_getFIFOLevel(uint32_t base, uint32_t *txLevel,
389 uint32_t *rxLevel)
390{
391 uint32_t temp;
392
393 //
394 // Check the arguments.
395 //
396 ASSERT(UART_isBaseValid(base));
397
398 //
399 // Read the FIFO level register.
400 //
401 temp = HWREG(base + UART_O_IFLS);
402
403 //
404 // Extract the transmit and receive FIFO levels.
405 //
406 *txLevel = temp & UART_IFLS_TXIFLSEL_M;
407 *rxLevel = temp & UART_IFLS_RXIFLSEL_M;
408}

480UART_disableModule(uint32_t base)
481{
482 //
483 // Check the arguments.
484 //
485 ASSERT(UART_isBaseValid(base));
486
487 //
488 // Wait for end of TX.
489 //
490 while((HWREG(base + UART_O_FR) & UART_FR_BUSY) == UART_FR_BUSY)
491 {
492 ;
493 }
494
495 //
496 // Disable the FIFO.
497 //
498 HWREG(base + UART_O_LCRH) &= ~(UART_LCRH_FEN);
499
500 //
501 // Disable the UART.
502 //
503 HWREG(base + UART_O_CTL) &= ~(UART_CTL_UARTEN | UART_CTL_TXE |
504 UART_CTL_RXE);
505}

841static inline int32_t
842UART_readCharNonBlocking(uint32_t base)
843{
844 int32_t ret;
845
846 //
847 // Check the arguments.
848 //
849 ASSERT(UART_isBaseValid(base));
850
851 //
852 // See if there are any characters in the receive FIFO.
853 //
854 if(((HWREG(base + UART_O_FR) & UART_FR_RXFE)) == 0U)
855 {
856 //
857 // Read and return the next character.
858 //
859 ret = (int32_t)HWREG(base + UART_O_DR);
860 }
861 else
862 {
863 //
864 // There are no characters, so return a failure.
865 //
866 ret = -1;
867 }
868 return(ret);
869}
