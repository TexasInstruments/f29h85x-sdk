275 static inline void
276 UART_setParityMode(uint32_t base, uint32_t parity)
277 {
278  //
279  // Check the arguments.
280  //
281  ASSERT(UART_isBaseValid(base));
282  ASSERT((parity == UART_CONFIG_PAR_NONE) ||
283  (parity == UART_CONFIG_PAR_EVEN) ||
284  (parity == UART_CONFIG_PAR_ODD) ||
285  (parity == UART_CONFIG_PAR_ONE) ||
286  (parity == UART_CONFIG_PAR_ZERO));
287 
288  //
289  // Set the parity mode.
290  //
291  HWREG(base + UART_O_LCRH) = ((HWREG(base + UART_O_LCRH) &
292  ~(UART_LCRH_SPS | UART_LCRH_EPS |
293  UART_LCRH_PEN)) | parity);
294 }

345 static inline void
346 UART_setFIFOLevel(uint32_t base, uint32_t txLevel,
347  uint32_t rxLevel)
348 {
349  //
350  // Check the arguments.
351  //
352  ASSERT(UART_isBaseValid(base));
353  ASSERT((txLevel == UART_FIFO_TX1_8) ||
354  (txLevel == UART_FIFO_TX2_8) ||
355  (txLevel == UART_FIFO_TX4_8) ||
356  (txLevel == UART_FIFO_TX6_8) ||
357  (txLevel == UART_FIFO_TX7_8));
358  ASSERT((rxLevel == UART_FIFO_RX1_8) ||
359  (rxLevel == UART_FIFO_RX2_8) ||
360  (rxLevel == UART_FIFO_RX4_8) ||
361  (rxLevel == UART_FIFO_RX6_8) ||
362  (rxLevel == UART_FIFO_RX7_8));
363 
364  //
365  // Set the FIFO interrupt levels.
366  //
367  HWREGB(base + UART_O_IFLS) = (uint8_t)(txLevel | rxLevel);
368 }

386 //
387 //*****************************************************************************
388 static inline void
389 UART_getFIFOLevel(uint32_t base, uint32_t *txLevel,
390  uint32_t *rxLevel)
391 {
392  uint32_t temp;
393 
394  //
395  // Check the arguments.
396  //
397  ASSERT(UART_isBaseValid(base));
398 
399  //
400  // Read the FIFO level register.
401  //
402  temp = HWREG(base + UART_O_IFLS);
403 
404  //
405  // Extract the transmit and receive FIFO levels.
406  //
407  *txLevel = temp & UART_IFLS_TXIFLSEL_M;
408  *rxLevel = temp & UART_IFLS_RXIFLSEL_M;
409 }
