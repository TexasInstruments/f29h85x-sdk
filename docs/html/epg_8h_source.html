498static inline void
499EPG_selectEPGOutput(uint32_t base, EPG_OUT gPinNum,
500 EPG_OutputSelect gPinOutSel)
501{
502 uint32_t regValue, bitFieldLoc, bitFieldVal;
503
504 ASSERT(EPG_isBaseValid(base));
505
506 bitFieldLoc = 1UL << ((uint16_t)gPinNum + EPG_GCTL0_EPGOUTSEL_S);
507 bitFieldVal = (uint32_t)gPinOutSel << ((uint16_t)gPinNum +
508 EPG_GCTL0_EPGOUTSEL_S);
509
510 regValue = (HWREG(base + EPG_O_GCTL0) & (~bitFieldLoc));
511 regValue |= bitFieldVal;
512
513 HWREG((base + EPG_O_GCTL0)) = regValue;
514}
