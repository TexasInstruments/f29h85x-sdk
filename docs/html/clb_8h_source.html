1140//*****************************************************************************
1141//
1161//
1162//*****************************************************************************
1163static inline void CLB_configLocalInputMux(uint32_t base, CLB_Inputs inID,
1164 CLB_LocalInputMux localMuxCfg)
1165{
1166 uint16_t shiftVal;
1167 uint32_t miscShiftVal, inputMuxSel;
1168
1169 ASSERT(CLB_isBaseValid(base));
1170
1171 //
1172 // Each local input has a 5-bit field in this register so need to calculate
1173 // shift amount accordingly.
1174 //
1175 shiftVal = (uint16_t)inID * CLB_LCL_MUX_SEL_1_LCL_MUX_SEL_IN_1_S;
1176
1177 //
1178 // If the local inputs are selected then only 5-bits are used in this
1179 // register but if miscellaneous inputs are selected then another 1-bit is
1180 // used along with the above 5. So we need to calculate this accordingly.
1181 //
1182
1183 //
1184 // Each miscellaneous input has a 1-bit field in this register so
1185 // need to calculate shift amount accordingly.
1186 //
1187 miscShiftVal = (uint32_t)inID + CLB_LCL_MUX_SEL_MISC_INPUT_SEL_S;
1188
1189 //
1190 // Check if input < input4
1191 //
1192 if(inID < CLB_IN4)
1193 {
1194 //
1195 // Each miscellaneous input has an extra 1-bit to be set
1196 // apart from the local input bits
1197 //
1198 inputMuxSel =
1199 ((((uint32_t)localMuxCfg & CLB_LCL_MUX_SEL_1_LCL_MUX_SEL_IN_0_M) <<
1200 shiftVal) |
1201 ((((uint32_t)localMuxCfg & CLB_LCL_MUX_SEL_MISC_INPUT_SEL_M) >>
1202 CLB_LCL_MUX_SEL_1_LCL_MUX_SEL_IN_1_S) << miscShiftVal));
1203
1204 //
1205 //Update the local / misc input mux selections
1206 //
1207 HWREG(base + CLB_LOGICCTL + CLB_O_LCL_MUX_SEL_1) =
1208 (HWREG(base + CLB_LOGICCTL + CLB_O_LCL_MUX_SEL_1) &
1209 ~(((uint32_t)CLB_LCL_MUX_SEL_1_LCL_MUX_SEL_IN_0_M << shiftVal) |
1210 (CLB_LCL_MUX_SEL_MISC_INPUT_SEL_BITM << miscShiftVal))) |
1211 inputMuxSel;
1212 }
1213 else
1214 {
1215 //
1216 //Adjust the previously calculated shifts for the inputs >3
1217 //
1218 shiftVal -= 4U * CLB_LCL_MUX_SEL_2_LCL_MUX_SEL_IN_5_S;
1219 miscShiftVal -= 4U;
1220
1221 //
1222 // Each miscellaneous input has an extra 1-bit to be set
1223 // apart from the local input bits
1224 //
1225 inputMuxSel =
1226 ((((uint32_t)localMuxCfg & CLB_LCL_MUX_SEL_2_LCL_MUX_SEL_IN_4_M) <<
1227 shiftVal) |
1228 ((((uint32_t)localMuxCfg & CLB_LCL_MUX_SEL_MISC_INPUT_SEL_M) >>
1229 CLB_LCL_MUX_SEL_2_LCL_MUX_SEL_IN_5_S) << miscShiftVal));
1230
1231 HWREG(base + CLB_LOGICCTL + CLB_O_LCL_MUX_SEL_2) =
1232 (HWREG(base + CLB_LOGICCTL + CLB_O_LCL_MUX_SEL_2) &
1233 ~(((uint32_t)CLB_LCL_MUX_SEL_2_LCL_MUX_SEL_IN_4_M << shiftVal) |
1234 (CLB_LCL_MUX_SEL_MISC_INPUT_SEL_BITM << miscShiftVal))) |
1235 inputMuxSel;
1236
1237 }
1238}
