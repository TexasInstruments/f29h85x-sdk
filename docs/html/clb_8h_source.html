1151 __attribute__((always_inline))
1152 static inline void CLB_configLocalInputMux(uint32_t base, CLB_Inputs inID,
1153  CLB_LocalInputMux localMuxCfg)
1154 {
1155  uint16_t shiftVal;
1156  uint32_t miscShiftVal, inputMuxSel;
1157 
1158  ASSERT(CLB_isBaseValid(base));
1159 
1160  //
1161  // Each local input has a 5-bit field in this register so need to calculate
1162  // shift amount accordingly.
1163  //
1164  shiftVal = (uint16_t)inID * CLB_LCL_MUX_SEL_1_LCL_MUX_SEL_IN_1_S;
1165 
1166  //
1167  // If the local inputs are selected then only 5-bits are used in this
1168  // register but if miscellaneous inputs are selected then another 1-bit is
1169  // used along with the above 5. So we need to calculate this accordingly.
1170  //
1171 
1172  //
1173  // Each miscellaneous input has a 1-bit field in this register so
1174  // need to calculate shift amount accordingly.
1175  //
1176  miscShiftVal = (uint32_t)inID + CLB_LCL_MUX_SEL_MISC_INPUT_SEL_S;
1177 
1178  //
1179  // Check if input < input4
1180  //
1181  if(inID < CLB_IN4)
1182  {
1183  //
1184  // Each miscellaneous input has an extra 1-bit to be set
1185  // apart from the local input bits
1186  //
1187  inputMuxSel =
1188  ((((uint32_t)localMuxCfg & CLB_LCL_MUX_SEL_1_LCL_MUX_SEL_IN_0_M) <<
1189  shiftVal) |
1190  ((((uint32_t)localMuxCfg & CLB_LCL_MUX_SEL_MISC_INPUT_SEL_M) >>
1191  CLB_LCL_MUX_SEL_1_LCL_MUX_SEL_IN_1_S) << miscShiftVal));
1192 
1193  //
1194  //Update the local / misc input mux selections
1195  //
1196  HWREG(base + CLB_LOGICCTL + CLB_O_LCL_MUX_SEL_1) =
1197  (HWREG(base + CLB_LOGICCTL + CLB_O_LCL_MUX_SEL_1) &
1198  ~(((uint32_t)CLB_LCL_MUX_SEL_1_LCL_MUX_SEL_IN_0_M << shiftVal) |
1199  (CLB_LCL_MUX_SEL_MISC_INPUT_SEL_BITM << miscShiftVal))) |
1200  inputMuxSel;
1201  }
1202  else
1203  {
1204  //
1205  //Adjust the previously calculated shifts for the inputs >3
1206  //
1207  shiftVal -= 4U * CLB_LCL_MUX_SEL_2_LCL_MUX_SEL_IN_5_S;
1208  miscShiftVal -= 4U;
1209 
1210  //
1211  // Each miscellaneous input has an extra 1-bit to be set
1212  // apart from the local input bits
1213  //
1214  inputMuxSel =
1215  ((((uint32_t)localMuxCfg & CLB_LCL_MUX_SEL_2_LCL_MUX_SEL_IN_4_M) <<
1216  shiftVal) |
1217  ((((uint32_t)localMuxCfg & CLB_LCL_MUX_SEL_MISC_INPUT_SEL_M) >>
1218  CLB_LCL_MUX_SEL_2_LCL_MUX_SEL_IN_5_S) << miscShiftVal));
1219 
1220  HWREG(base + CLB_LOGICCTL + CLB_O_LCL_MUX_SEL_2) =
