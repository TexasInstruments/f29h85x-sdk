500 static inline void
501 EMIF_setAsyncTimingParams(uint32_t base, EMIF_AsyncCSOffset offset,
502  const EMIF_AsyncTimingParams *tParam)
503 {
504  uint32_t temp;
505  //
506  // Check the arguments.
507  //
508  ASSERT(EMIF_isBaseValid(base));
509 
510  //
511  // Sets the async memory timing parameters.
512  //
513  temp = (tParam->turnArnd << EMIF_ASYNC_CS2_CR_TA_S) |
514  (tParam->rHold << EMIF_ASYNC_CS2_CR_R_HOLD_S) |
515  (tParam->rStrobe << EMIF_ASYNC_CS2_CR_R_STROBE_S) |
516  (tParam->rSetup << EMIF_ASYNC_CS2_CR_R_SETUP_S) |
517  (tParam->wHold << EMIF_ASYNC_CS2_CR_W_HOLD_S) |
518  (tParam->wStrobe << EMIF_ASYNC_CS2_CR_W_STROBE_S) |
519  (tParam->wSetup << EMIF_ASYNC_CS2_CR_W_SETUP_S);
520 
521  HWREG(base + (uint32_t)offset) = (HWREG(base + (uint32_t)offset) &
522  ~EMIF_ASYNC_CS_CR_MASK) | temp;
523 }

733  | ((tParam->tRp << EMIF_SDRAM_TR_T_RP_S)
734  & EMIF_SDRAM_TR_T_RP_M)
735  | ((tParam->tRfc << EMIF_SDRAM_TR_T_RFC_S)
736  & EMIF_SDRAM_TR_T_RFC_M);
