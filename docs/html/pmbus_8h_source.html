690 __attribute__((always_inline))
691 static inline void PMBus_configController(uint32_t base, uint16_t targetAddress,
692  uint16_t byteCount, uint32_t configWord)
693 {
694  //
695  // Locals
696  // Form a bit mask of the bit fields configWord changes
697  //
698  const uint32_t bitmask = (PMBUS_PMBCCR_RW | PMBUS_PMBCCR_CMD_ENA |
699  PMBUS_PMBCCR_EXT_CMD | PMBUS_PMBCCR_PEC_ENA |
700  PMBUS_PMBCCR_GRP_CMD | PMBUS_PMBCCR_PRC_CALL);
701 
702  //
703  // Check the arguments.
704  //
705  ASSERT(PMBus_isBaseValid(base));
706  ASSERT(targetAddress <= 0x7FU);
707  ASSERT(byteCount <= 0xFFU);
708 
709 
710  //
711  // Write the user configured bit fields (passed in configWord) as well
712  // as target address and message byte counts. Writing to PMBMC triggers
713  // activity on the bus, once the bus is free (PMBSTS.BUS_FREE = 1)
714  //
715  HWREG(base + PMBUS_O_PMBCCR) = ((configWord & bitmask) |
716  (((uint32_t)targetAddress << PMBUS_PMBCCR_TARGET_ADDR_S) &
717  PMBUS_PMBCCR_TARGET_ADDR_M) |
718  (((uint32_t)byteCount << PMBUS_PMBCCR_BYTE_COUNT_S) &
719  PMBUS_PMBCCR_BYTE_COUNT_M));
720 }
