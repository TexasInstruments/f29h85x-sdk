478static inline void PMBus_enableI2CMode(uint32_t base)
479{
480 //
481 // Locals
482 //
483 uint32_t interruptState;
484
485 //
486 // Check the arguments.
487 //
488 ASSERT(PMBus_isBaseValid(base));
489
490 //
491 // Save off the interrupt state and disable them
492 //
493 interruptState = HWREG(base + PMBUS_O_PMBINTM);
494 HWREG(base + PMBUS_O_PMBINTM) = PMBUS_INT_ALL;
495
496 //
497 // Set module to I2C mode
498 //
499 HWREG(base + PMBUS_O_PMBCTRL) |= PMBUS_PMBCTRL_I2CMODE;
500
501 //
502 // Restore the interrupt status
503 //
504 HWREG(base + PMBUS_O_PMBINTM) = interruptState;
505
506}

662//*****************************************************************************
663//
706//
707//*****************************************************************************
708static inline void PMBus_configController(uint32_t base, uint16_t targetAddress,
709 uint16_t byteCount, uint32_t configWord)
710{
711 //
712 // Locals
713 // Form a bit mask of the bit fields configWord changes
714 //
715 const uint32_t bitmask = (PMBUS_PMBCCR_RW | PMBUS_PMBCCR_CMD_ENA |
716 PMBUS_PMBCCR_EXT_CMD | PMBUS_PMBCCR_PEC_ENA |
717 PMBUS_PMBCCR_GRP_CMD | PMBUS_PMBCCR_PRC_CALL);
718
719 //
720 // Check the arguments.
721 //
722 ASSERT(PMBus_isBaseValid(base));
723 ASSERT(targetAddress <= 0x7FU);
724 ASSERT(byteCount <= 0xFFU);
725
726
727 //
728 // Write the user configured bit fields (passed in configWord) as well
729 // as target address and message byte counts. Writing to PMBMC triggers
730 // activity on the bus, once the bus is free (PMBSTS.BUS_FREE = 1)
731 //
732 HWREG(base + PMBUS_O_PMBCCR) = ((configWord & bitmask) |
733 (((uint32_t)targetAddress << PMBUS_PMBCCR_TARGET_ADDR_S) &
734 PMBUS_PMBCCR_TARGET_ADDR_M) |
735 (((uint32_t)byteCount << PMBUS_PMBCCR_BYTE_COUNT_S) &
736 PMBUS_PMBCCR_BYTE_COUNT_M));
737}

865 //
