170 static inline void
171 SysCtl_setWatchdogMode(SysCtl_WDMode mode)
172 {
173  //
174  // Either set or clear the WDENINT bit to that will determine whether the
175  // watchdog will generate a reset signal or an interrupt signal. Take care
176  // not to write a 1 to WDOVERRIDE.
177  //
178  if(mode == SYSCTL_WD_MODE_INTERRUPT)
179  {
180  HWREGH(WD_BASE + SYSCTL_O_SCSR) =
181  (HWREGH(WD_BASE + SYSCTL_O_SCSR) & ~SYSCTL_SCSR_WDOVERRIDE) |
182  SYSCTL_SCSR_WDENINT;
183  }
184  else
185  {
186  HWREGH(WD_BASE + SYSCTL_O_SCSR) &= ~(SYSCTL_SCSR_WDENINT |
187  SYSCTL_SCSR_WDOVERRIDE);
188  }
189 
190 }
