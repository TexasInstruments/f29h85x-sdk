141 __attribute__((always_inline))
142 static inline void
143 SysCtl_setWatchdogMode(SysCtl_WDMode mode)
144 {
145  //
146  // Either set or clear the WDENINT bit to that will determine whether the
147  // watchdog will generate a reset signal or an interrupt signal. Take care
148  // not to write a 1 to WDOVERRIDE.
149  //
150  if(mode == SYSCTL_WD_MODE_INTERRUPT)
151  {
152  HWREGH(WD_BASE + SYSCTL_O_SCSR) =
153  (HWREGH(WD_BASE + SYSCTL_O_SCSR) & ~SYSCTL_SCSR_WDOVERRIDE) |
154  SYSCTL_SCSR_WDENINT;
155  }
156  else
157  {
158  HWREGH(WD_BASE + SYSCTL_O_SCSR) &= ~(SYSCTL_SCSR_WDENINT |
159  SYSCTL_SCSR_WDOVERRIDE);
160  }
161 
162 }
