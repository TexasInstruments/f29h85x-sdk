173static inline void
174SysCtl_setWatchdogMode(SysCtl_WDMode mode)
175{
176 //
177 // Either set or clear the WDENINT bit to that will determine whether the
178 // watchdog will generate a reset signal or an interrupt signal. Take care
179 // not to write a 1 to WDOVERRIDE.
180 //
181 if(mode == SYSCTL_WD_MODE_INTERRUPT)
182 {
183 HWREGH(WD_BASE + SYSCTL_O_SCSR) =
184 (HWREGH(WD_BASE + SYSCTL_O_SCSR) & ~SYSCTL_SCSR_WDOVERRIDE) |
185 SYSCTL_SCSR_WDENINT;
186 }
187 else
188 {
189 HWREGH(WD_BASE + SYSCTL_O_SCSR) &= ~(SYSCTL_SCSR_WDENINT |
190 SYSCTL_SCSR_WDOVERRIDE);
191 }
192
193}

212static inline bool
213SysCtl_isWatchdogInterruptActive(void)
214{
215 //
216 // If the status bit is cleared, the WDINTn signal is active.
217 //
218 return((HWREGH(WD_BASE + SYSCTL_O_SCSR) & SYSCTL_SCSR_WDINTS) == 0U);
219}

230//*****************************************************************************
231static inline void
232SysCtl_disableWatchdog(void)
233{
234 //
235 // Set the disable bit.
236 //
237 HWREGH(WD_BASE + SYSCTL_O_WDCR) |= SYSCTL_WD_CHKBITS | SYSCTL_WDCR_WDDIS;
238
239 //
240 // Poll until WD sync busy is cleared
241 //
242 SysCtl_pollSyncBusyWd(SYSCTL_SYNCBUSYWD_WDCR);
243}

254//*****************************************************************************
255static inline void
256SysCtl_enableWatchdog(void)
257{
258 //
259 // Clear the disable bit.
260 //
261 HWREGH(WD_BASE + SYSCTL_O_WDCR) = (HWREGH(WD_BASE + SYSCTL_O_WDCR) &
262 ~SYSCTL_WDCR_WDDIS) | SYSCTL_WD_CHKBITS;
263
264 //
265 // Poll until WD sync busy is cleared
266 //
267 SysCtl_pollSyncBusyWd(SYSCTL_SYNCBUSYWD_WDCR);
268}

339static inline void
340SysCtl_setWatchdogPredivider(SysCtl_WDPredivider predivider)
341{
342 uint16_t regVal;
343
344 regVal = (uint16_t)predivider | SYSCTL_WD_CHKBITS;
345
346 //
347 // Write the predivider to the appropriate register.
348 //
349 HWREGH(WD_BASE + SYSCTL_O_WDCR) = (HWREGH(WD_BASE + SYSCTL_O_WDCR) &
350 ~(SYSCTL_WDCR_WDPRECLKDIV_M)) | regVal;
351
352 //
353 // Poll until WD sync busy is cleared
354 //
355 SysCtl_pollSyncBusyWd(SYSCTL_SYNCBUSYWD_WDCR);
356}
