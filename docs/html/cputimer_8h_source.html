173//*****************************************************************************
174//
182//
183//*****************************************************************************
184static inline void CPUTimer_enableInterrupt(uint32_t base)
185{
186 uint16_t tcrValue = 0;
187 ASSERT(CPUTimer_isBaseValid(base));
188
189 //
190 // Set TIE bit of TCR register
191 //
192 tcrValue = HWREGH(base + CPUTIMER_O_TCR) & (~CPUTIMER_TCR_TIF);
193 HWREGH(base + CPUTIMER_O_TCR) = tcrValue | CPUTIMER_TCR_TIE;
194}

277//
278//*****************************************************************************
279static inline void CPUTimer_startTimer(uint32_t base)
280{
281 uint16_t tcrValue = 0;
282 ASSERT(CPUTimer_isBaseValid(base));
283
284 //
285 // Reload the timer counter
286 //
287 tcrValue = HWREGH(base + CPUTIMER_O_TCR) & (~CPUTIMER_TCR_TIF);
288 HWREGH(base + CPUTIMER_O_TCR) = tcrValue | CPUTIMER_TCR_TRB;
289
290 //
291 // Clear TSS bit of register TCR
292 //
293 HWREGH(base + CPUTIMER_O_TCR) &= ~CPUTIMER_TCR_TSS;
294}

308static inline void CPUTimer_setPeriod(uint32_t base, uint32_t periodCount)
309{
310 ASSERT(CPUTimer_isBaseValid(base));
311 ASSERT(periodCount > 1U);
312
313 //
314 // Load the MSB period Count
315 //
316 HWREG(base + CPUTIMER_O_PRD) = (periodCount - 1U);
317}

353static inline void CPUTimer_setPreScaler(uint32_t base, uint16_t prescaler)
354{
355 ASSERT(CPUTimer_isBaseValid(base));
356
357 //
358 // Writes to TPR.TDDR and TPRH.TDDRH bits
359 //
360 HWREGH(base + CPUTIMER_O_TPRH) = prescaler >> 8U;
361 HWREGH(base + CPUTIMER_O_TPR) = (prescaler & CPUTIMER_TPR_TDDR_M) ;
362}

375static inline bool CPUTimer_getTimerOverflowStatus(uint32_t base)
376{
377 ASSERT(CPUTimer_isBaseValid(base));
378
379 //
380 // Check if TIF bits of register TCR are set
381 //
382 return(((HWREGH(base + CPUTIMER_O_TCR) & CPUTIMER_TCR_TIF) ==
383 CPUTIMER_TCR_TIF) ? true : false);
384}
