143/* If building for R5 then read selfcore ID from the Core Cluster register */
144static uint8_t SIPC_readSelfCoreID(void)
145{
146 uint8_t coreId;
147 coreId = CpuIdP_c29GetCpuID();

149 /* both cpuId and greId fields can be either 0 or 1
150 * ex for R5FSS1-0 core grpId = 1 and cpu id = 0 */
151 if(1 == coreId){
152 coreId = CORE_ID_C29_CPU_1;
153 }
154 else if(3 == coreId){
155 coreId = CORE_ID_C29_CPU_3;
156 }
