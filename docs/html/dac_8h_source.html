276 // Set the PWM sync signal

278

280 ~DAC_CTL_SYNCSEL_M) |
281 ((uint16_t)(pwmSignal - 1U) <<
282 DAC_CTL_SYNCSEL_S);

464static inline int16_t
465DAC_getOffsetTrim(uint32_t base)
466{
467 uint16_t value;
468
469 //
470 // Check the arguments.
471 //
472 ASSERT(DAC_isBaseValid(base));
473
474 //
475 // Get the sign-extended offset trim value
476 //
477 value = (HWREGH(base + DAC_O_TRIM) & DAC_TRIM_OFFSET_TRIM_M);
478 value = ((value & (uint16_t)DAC_REG_BYTE_MASK) ^ 0x80U) -
479 0x80U;
480
481 return((int16_t)value);
482}

503static inline void
504DAC_lockRegister(uint32_t base, uint16_t reg)
505{
506 //
507 // Check the arguments.
508 //
509 ASSERT(DAC_isBaseValid(base));
510 ASSERT((reg & ~(DAC_LOCK_CONTROL | DAC_LOCK_SHADOW |
511 DAC_LOCK_OUTPUT)) == 0U);
512
513 //
514 // Lock the specified registers
515 //
516
517 HWREGH(base + DAC_O_LOCK) |= (DAC_LOCK_KEY | reg);
518}
