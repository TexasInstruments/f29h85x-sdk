308__attribute__((always_inline))
309static inline bool SDL_ESM_isBaseValid(uint32_t esmBase)
310{
311 return(
312 ((esmBase & SDL_BASE_ADDR_MASK) == SDL_ESMCPU1_BASE_FRAME(0U)) ||
313 ((esmBase & SDL_BASE_ADDR_MASK) == SDL_ESMCPU2_BASE_FRAME(0U)) ||
314 ((esmBase & SDL_BASE_ADDR_MASK) == SDL_ESMCPU3_BASE_FRAME(0U)) ||
315 ((esmBase & SDL_BASE_ADDR_MASK) == SDL_ESMSYSTEM_BASE_FRAME(0U))
316 );
317}
