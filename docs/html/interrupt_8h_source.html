352static inline void
353Interrupt_register(uint32_t intNum, void (*handler)(void))
354{
355 //
356 // INT_NMI used as an id and has no numeric significance
357 //
358 ASSERT((intNum < INTERRUPT_NO_OF_CHANNELS) || (intNum == INT_NMI));
359
360 if(intNum == INT_NMI)
361 {
362 HWREG(PIPE_BASE + PIPE_O_NMI_VECT) = (uint32_t)handler;
363 }
364 else
365 {
366 HWREG(PIPE_BASE + PIPE_O_INT_VECT_ADDR(intNum)) = (uint32_t)handler;
367 }
368}

481//*****************************************************************************
482//
489//
490//*****************************************************************************
491static inline void
492Interrupt_setLinkOwner(uint32_t intNum, SSU_Link linkOwner)
493{
494 //
495 // INT_NMI used as an id and has no numeric significance
496 //
497 ASSERT((intNum < INTERRUPT_NO_OF_CHANNELS) || (intNum == INT_NMI));
498
499 if(intNum == INT_NMI)
500 {
501 HWREGB(PIPE_BASE + PIPE_O_NMI_LINK_OWNER) = (uint8_t)linkOwner;
502 }
503 else
504 {
505 HWREGB(PIPE_BASE + PIPE_O_INT_LINK_OWNER(intNum)) =
506 (HWREGB(PIPE_BASE + PIPE_O_INT_LINK_OWNER(intNum)) & 0xF0U) |
507 (uint8_t)linkOwner;
508 }
509}

511//*****************************************************************************
512//
520//
521//*****************************************************************************
522static inline void
523Interrupt_setAPILink(uint32_t intNum, SSU_APILink apiLinkID)
524{
525 ASSERT(intNum < INTERRUPT_NO_OF_CHANNELS);
526
527 if(apiLinkID == SSU_API_DISABLE)
528 {
529 HWREGH(PIPE_BASE + PIPE_O_INT_LINK_OWNER(intNum)) &= 0xFU;
530 }
531 else
532 {
533 HWREGH(PIPE_BASE + PIPE_O_INT_LINK_OWNER(intNum)) =
534 (HWREGB(PIPE_BASE + PIPE_O_INT_LINK_OWNER(intNum)) & 0x0FU) |
535 PIPE_INT_LINK_OWNER_API_LINK_EN |
536 ((uint16_t)apiLinkID << PIPE_INT_LINK_OWNER_API_LINK_S);
537 }
538}
