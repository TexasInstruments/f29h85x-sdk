523 
524  if(apiLinkID == SSU_API_DISABLE)

527  }

529  {
530  HWREGH(PIPE_BASE + PIPE_O_INT_LINK_OWNER(intNum)) =
531  (HWREGB(PIPE_BASE + PIPE_O_INT_LINK_OWNER(intNum)) & 0x0FU) |
532  PIPE_INT_LINK_OWNER_API_LINK_EN |
533  ((uint16_t)apiLinkID << PIPE_INT_LINK_OWNER_API_LINK_S);
534  }
